```c
#include<stdio.h>
int main(){
	int m,t,s,less;
	scanf("%d%d%d",&m,&t,&s);
	if(t!=0) less=m-(s+t-1)/t;
	if(t==0 &&s==0) less=m;
	if(t==0 &&s>0) less=0;
	printf("%d",less>=0?less:0);
	return 0;
} 
```

第二题

```c
#include<stdio.h>
int main(){
	int m,s,a,b,c,d;
	s=0;
	scanf("%d",&m);
	if(m%2==0) s++;
	if(m > 4 && m <= 12) s++;
	switch(s){
		case 0:
            a=0,b=0,c=0,d=1;
            break;
		case 1:
            a=0,b=1,c=1,d=0;
            break;
		case 2:
            a=1,b=1,c=0,d=0;
            break;
        default:
            a=0,b=0,c=0,d=0;
            break;
	} 
	printf("%d %d %d %d",a,b,c,d);
	return 0;
} 
```

评价：特别的思路，但是有一点点小题大做

答案：

```c
#include <stdio.h>

int main(){
    float m , h ;

    scanf("%f%f",&m,&h);

    float bmi = m/(h*h);

    if(bmi < 18.5) printf("Underweight");
    else if(bmi < 24.5) printf("Normal");
    else printf("%.4f\nOverweight",bmi);

    return 0;
}
```
